SELECT I.NAME, COUNT(*) AS LIST
FROM ISSUESLISTS I JOIN ISSUES P on I.ID = P.ISSUESLIST_ID
GROUP BY I.NAME;

SELECT SUMMARY, DESCRIPTION,
       USER_ID_ASSIGNEDTO FROM ISSUES;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS LIST
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME;
